class Animal {

  public void eat() {
    System.out.println("I can eat");
  }
}

class Cat extends Animal 

{
 public void eat() {
    System.out.println("I eat Cat food");
  }

  public void bark() {
    System.out.println("I can bark");
  }
}

class Main {
  public static void main(String[] args) {
    Cat labrador = new Cat();
    labrador.eat();
    labrador.bark();
  }
}
